function calcularComisionSocio(params: {
  plan: 'Impulsa' | 'Transforma' | 'Lidera',
  numColaboradores: number,
  numEmpresasActivas: number, // para determinar % de comisión
  tipoSocio: 'consultor' | 'director',
  numConsultoresEnZona?: number, // solo para director
  empresasDeConsultores?: number, // solo para director override
}) {
  const UMA = 113.14;
  const PLAN_UMAS = { Impulsa: 6, Transforma: 10, Lidera: 20 };
  const PLAN_FEE_PCT = { Impulsa: 0.15, Transforma: 0.08, Lidera: 0.05 };
  
  // Aportación mensual de la empresa
  const aportacionMensual = numColaboradores * PLAN_UMAS[plan] * UMA;
  
  // Fee de Ceduverse
  const feeCeduverse = aportacionMensual * PLAN_FEE_PCT[plan];
  
  // % de comisión según cartera
  let comisionPct;
  if (numEmpresasActivas <= 3) comisionPct = 0.25;
  else if (numEmpresasActivas <= 7) comisionPct = 0.30;
  else comisionPct = 0.35;
  
  // Comisión residual mensual
  const comisionResidual = feeCeduverse * comisionPct;
  
  // Comisión de cierre (primer mes)
  const comisionCierre = feeCeduverse * 0.50;
  
  // Certificaciones estimadas anuales
  const dc3Estimados = Math.floor(numColaboradores * 0.40); // 40% compra DC3
  const sepEstimados = Math.floor(numColaboradores * 0.05); // 5% compra SEP
  const ingresoCertsAnual = (dc3Estimados * 60) + (sepEstimados * 200);
  
  // Override director
  let overrideMensual = 0;
  if (tipoSocio === 'director' && empresasDeConsultores) {
    // 5% del fee de empresas de los consultores
    // Estimación: empresas similares al promedio de la zona
    overrideMensual = feeCeduverse * 0.05 * empresasDeConsultores;
  }
  
  return {
    aportacionMensual,
    feeCeduverse,
    comisionPct,
    comisionResidual,
    comisionCierre,
    dc3Estimados,
    sepEstimados,
    ingresoCertsAnual,
    overrideMensual,
    ingresoAnualTotal: (comisionResidual * 12) + ingresoCertsAnual + (overrideMensual * 12),
  };
}
